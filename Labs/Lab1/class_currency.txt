class Currency {
	private String _currency;
	public Currency( String currency) {
		setCurrency(currency);
	}
	public boolean setCurrency( String currency) {
		if(!(currency instanceof String)) {
			return false;
		}
		else if(currency.length() != 3) {
			return false;
		}
		else {
			this._currency = currency;
			return true;
		}
		
	}
}
